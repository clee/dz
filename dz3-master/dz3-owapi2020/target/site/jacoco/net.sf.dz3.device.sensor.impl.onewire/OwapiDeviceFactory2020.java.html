package net.sf.dz3.device.sensor.impl.onewire;

import java.util.Enumeration;
import java.util.LinkedHashMap;
import java.util.Map;
import java.util.Map.Entry;
import java.util.Set;
import java.util.TreeSet;
import java.util.concurrent.Future;

import org.apache.logging.log4j.LogManager;
import org.apache.logging.log4j.Logger;

import com.dalsemi.onewire.OneWireAccessProvider;
import com.dalsemi.onewire.OneWireException;
import com.dalsemi.onewire.adapter.DSPortAdapter;
import com.dalsemi.onewire.adapter.DSPortAdapter.Reset;
import com.dalsemi.onewire.adapter.DSPortAdapter.Speed;

import net.sf.dz3.device.actuator.PowerOffCapable;
import net.sf.dz3.device.sensor.AnalogSensor;
import net.sf.dz3.device.sensor.DeviceFactory2020;
import net.sf.dz3.device.sensor.Switch;

/**
 * 1-Wire device factory.
 *
 * The [simpler] descendant of {@link OwapiDeviceFactory}, implemented 20 years later.
 *
 * @author Copyright &copy; <a href="mailto:vt@homeclimatecontrol.com">Vadim Tkachenko</a> 2001-2020
 */
public class OwapiDeviceFactory2020 implements DeviceFactory2020, PowerOffCapable<Void> {

    protected final Logger logger = LogManager.getLogger(getClass());

    private final BusAdapter busAdapter;

    @SuppressWarnings("FutureReturnValueIgnored")
    public OwapiDeviceFactory2020(String port, String speed) throws OneWireException, InterruptedException {

        Speed s = Speed.valueFor(speed);
        Map<DSPortAdapter, Set<String>> adapterMap = getAdapters();

        logger.debug("adapters: {}", adapterMap);

        DSPortAdapter adapter = resolve(adapterMap, port);

        logger.info("adapter: {}", adapter.getAdapterName());

        select(adapter, port, adapterMap);

        logger.debug("reset: {}", Reset.valueOf(adapter.reset()));
        adapter.setSpeed(s.value);

        logger.debug("speed: {}",  s);

        busAdapter = startAdapter(adapter, s);

        // Need to wait until browse() at least starts

        // VT: NOTE: Future value is not ignored; the point of this call is to make sure
        // it's been created

        busAdapter.getBrowseFuture();
    }

    void select(DSPortAdapter a, String port, Map<DSPortAdapter, Set<String>> adapterMap) throws OneWireException {
        if (!a.selectPort(port)) {
            throw new IllegalArgumentException("can't select port '" + port + "', possible combinations: " + adapterMap);
        }
    }

    private Map<DSPortAdapter, Set<String>> getAdapters() {

        // DSPortAdapter is not Comparable
        Map<DSPortAdapter, Set<String>> result = new LinkedHashMap<>();

        for (Enumeration<DSPortAdapter> e = OneWireAccessProvider
                .enumerateAllAdapters(); e.hasMoreElements();) {

            DSPortAdapter a = e.nextElement();

            Set<String> ports = new TreeSet<>();

            for (Enumeration<String> e2 = a.getPortNames(); e2.hasMoreElements();) {
                ports.add(e2.nextElement());
            }

            result.put(a, ports);
        }

        return result;
    }

    private DSPortAdapter resolve(Map<DSPortAdapter, Set<String>> adapterMap, String port) {

        for (Entry<DSPortAdapter, Set<String>> ap : adapterMap.entrySet()) {
            if (ap.getValue().contains(port)) {
                return ap.getKey();
            }
        }

        throw new IllegalArgumentException("can't find an adapter for port '" + port + "', possible combinations: " + adapterMap);
    }

    /**
     * Create and start the adapter.
     *
     * We will never stop it, even on {@link #powerOff()}.
     *
     * We don't care about the fact that it's running in its own thread, either.
     *
     * @param adapter Adapter instance to work with.
     * @param speed Adapter speed, in case we need to reset it.
     *
     * @return The instance of the adapter.
     */
    private BusAdapter startAdapter(DSPortAdapter adapter, Speed speed) {

        BusAdapter ba = new BusAdapter(adapter, speed);

        new Thread(ba).start();

        return ba;
    }

    @Override
    public AnalogSensor getSensor(String address) {
        return busAdapter.getSensor(address);
    }

    @Override
    public Switch getSwitch(String address) {
        return busAdapter.getSwitch(address);
    }

    @Override
    public void powerOff() {
        logger.warn("powerOff()");
        try {
            busAdapter.powerOff();
        } catch (Throwable t) {
            logger.fatal("failed to shut down cleanly, nothing we can do here", t);
        }
    }

    @Override
    public Future<Void> powerOffAsync() {
        throw new UnsupportedOperationException("Not Implemented");
    }
}
