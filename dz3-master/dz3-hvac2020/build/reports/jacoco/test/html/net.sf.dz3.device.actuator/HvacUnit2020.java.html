package net.sf.dz3.device.actuator;

import java.util.concurrent.Future;

import net.sf.dz3.device.model.HvacMode;
import com.homeclimatecontrol.jukebox.datastream.signal.model.DataSink;
import com.homeclimatecontrol.jukebox.jmx.JmxAttribute;
import com.homeclimatecontrol.jukebox.jmx.JmxAware;

/**
 * The 2020 abstraction for HVAC units.
 *
 * @author Copyright &copy; <a href="mailto:vt@homeclimatecontrol.com">Vadim Tkachenko</a> 2001-2020
 */
public interface HvacUnit2020 extends JmxAware, DataSink<Double>, Comparable<HvacUnit2020>, PowerOffCapable<HvacUnit2020.Status> {

    public static class Status {

        public final HvacMode mode;
        public final double demand;
        public final double fanSpeed;

        public Status(HvacMode mode, double demand, double fanSpeed) {
            this.mode = mode;
            this.demand = demand;
            this.fanSpeed = fanSpeed;
        }
    }

    /**
     * Get the name of the unit.
     */
    @JmxAttribute(description="Unit name")
    String getName();

    /**
     * @return Requested status.
     */
    @JmxAttribute(description="Current status (requested)")
    Status getStatus();

    /**
     * @return Actual status. May be different from {@link #getStatus()} due to hardware delays.
     */
    @JmxAttribute(description="Current status (actual)")
    Status getActualStatus();

    /**
     * Set operating mode.
     *
     * @param mode Mode to set.
     * @return The future that becomes done when the hardware completes the operation.
     */
    Future<Status> setMode(HvacMode mode);

    /**
     * Set the unit demand.
     *
     * Any value greater than {@code 0} will cause the unit to switch on.
     *
     * Any value equal or less than {@code 0} will cause the unit to switch off.
     *
     * This is true for both heating and cooling mode; it is up to the
     * implementation how to translate the demand into hardware instructions.
     *
     * @param demand Demand to set.
     * @return The future that becomes done when the hardware completes the operation.
     */
    Future<Status> setDemand(double demand);

    /**
     * Set the fan speed.
     *
     * Any value greater than {@code 0} will cause the fan to switch on.
     *
     * Any value equal or less than {@code 0} will cause the unit to switch off.
     *
     * Any value greater than {@code 1} is considered a programming error and
     * must cause the implementation to throw an {@code IllegalArgumentException}.
     *
     * This is true for both heating and cooling mode; it is up to the
     * implementation how to translate the demand into hardware instructions.
     *
     * @param demand Demand to set.
     * @return The future that becomes done when the hardware completes the operation.
     */
    Future<Status> setFanSpeed(double speed);

    /**
     * Power off the device in a safe way, synchronously.
     *
     * NOTE: This method is provided to provide background compatibility with the
     * synchronous {@link HvacDriver#powerOff()}, use {@link #powerOffAsync()}
     * instead if possible.
     */
    @Override
    void powerOff();

    /**
     * Power off the device in a safe way, asynchronously.
     *
     * NOTE: The method specification is asynchronous. Extreme caution must be paid by the caller
     * not to shut down the JVM before the operation is complete.
     *
     * @return The future that becomes done when the hardware completes the operation.
     */
    @Override
    Future<Status> powerOffAsync();
}
