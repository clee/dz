package net.sf.dz3.device.actuator.impl;

import java.util.SortedMap;
import java.util.concurrent.Callable;
import java.util.concurrent.ExecutionException;
import java.util.concurrent.Future;

import org.apache.logging.log4j.ThreadContext;

import net.sf.dz3.device.model.HvacMode;
import net.sf.dz3.device.sensor.Switch;
import com.homeclimatecontrol.jukebox.jmx.JmxDescriptor;

public class HeatPump extends AbstractHvacUnit2020 {

    private final Switch switchMode;
    private SortedMap<Double, Switch> demandMap;
    private SortedMap<Double, Switch> speedMap;

    /**
     * Create a multistage instance.
     *
     * @param name Heatpump name.
     * @param mode Initial mode.
     *
     * @param switchMode Switch to use to change mode. It is assumed {@code off} is for cooling, and {@code on} is for heating,
     * reverse it outside if necessary.
     *
     * @param demandMap The values are demand levels to turn the particular switch on and off at, keys are switches to activate.
     * The caller is supposed to take care of the hysteresis to avoid the switch jitter.
     * Demand values will be normalized to fit into {@code 0..1} range.
     *
     * @param speedMap The values are fan speed levels to turn the particular switch on and off at, keys are switches to activate.
     * The caller is supposed to take care of the hysteresis to avoid the switch jitter.
     * Speed values will be normalized to fit into {@code 0..1} range.
     */
    public HeatPump(String name, HvacMode mode,
            Switch switchMode,
            SortedMap<Double, Switch> demandMap,
            SortedMap<Double, Switch> speedMap) {

        super(name, mode);

        this.switchMode = switchMode;
        this.demandMap = normalize(demandMap);
        this.speedMap = normalize(speedMap);

        // The mode has to be set correctly from the get go.
        // If we can't, fail fast - this is a catastrophic event.

        try {
            set(mode, 0, 0).get();
        } catch (InterruptedException | ExecutionException ex) {
            throw new IllegalStateException("can't set initial mode", ex);
        }
    }

    /**
     * Create a single switch instance.
     *
     * @param s The only switch to control the hardware. An entry of
     * {@code (theSwitch, 0)} will be put into {@link #demandMap}.
     */
    public HeatPump(String name, HvacMode mode,
            Switch switchMode,
            Switch switchDemand,
            Switch switchFan) {

        this(name, mode,
                switchMode,
                normalize(map(switchDemand)),
                normalize(map(switchFan)));
    }

    @Override
    public JmxDescriptor getJmxDescriptor() {

        return new JmxDescriptor(
                "dz",
                "Heatpump: " + getName(),
                Integer.toHexString(hashCode()),
                "Controls a heatpump");
    }

    @Override
    public String toString() {

        // The only essential parts (for now) are the name, the demand map, and the speed map,
        // everything else is secondary

        // VT: NOTE: It might be a good idea to package this as JSON
        return getClass().getSimpleName()
                + ": {name:" + getName()
                + ", demandMap:" + demandMap
                + ", speedMap:" + speedMap
                + "}";
    }

    @Override
    protected Future<Status> set(HvacMode mode, double demand, double speed) {
        Callable<Status> c = new Callable<Status>() {

            @Override
            public Status call() throws Exception {
                ThreadContext.push("set");
                try {

                    // It is assumed that the device is "energize to heat", if this is not the case,
                    // invert the switch passed to the constructor
                    boolean modeState = mode == HvacMode.HEATING;

                    // It'll take a bit of time, but it won't hurt because we'll be providing a nice side effect - heartbeats
                    switchMode.setState(modeState);

                    execute(demand, demandMap);
                    execute(map(demand, speed), speedMap);

                    Status statusActual = new Status(mode, demand, speed);

                    setStatusActual(statusActual);

                    return statusActual;

                } finally {
                    ThreadContext.pop();
                }
            }
        };

        return executor.submit(c);
    }

    @Override
    protected void verifyModeChange(HvacMode mode) {
        // Nothing to do here, all modes are good
    }
}
