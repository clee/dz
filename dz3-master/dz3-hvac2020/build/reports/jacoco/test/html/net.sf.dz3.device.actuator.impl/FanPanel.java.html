package net.sf.dz3.device.actuator.impl;

import java.util.SortedMap;
import java.util.concurrent.Callable;
import java.util.concurrent.Future;

import org.apache.logging.log4j.ThreadContext;

import net.sf.dz3.device.model.HvacMode;
import net.sf.dz3.device.model.ZoneController;
import net.sf.dz3.device.sensor.Switch;
import com.homeclimatecontrol.jukebox.jmx.JmxDescriptor;

public class FanPanel extends AbstractHvacUnit2020 {

    private SortedMap<Double, Switch> demandMap;

    /**
     * Create a multi switch instance.
     *
     * @param name Fan panel name.
     * @param mode Mode reported by this instance, for compatibility. An attempt to {@link #setMode(HvacMode)}
     * to a mode different than this (other than {@link HvacMode#OFF} will cause an {@link IllegalArgumentException}.
     *
     * @param demandMap The values are demand levels to turn the particular switch on and off at, keys are switches to activate.
     * The caller is supposed to take care of the hysteresis to avoid the switch jitter.
     * Demand values will be normalized to fit into {@code 0..1} range.
     */
    public FanPanel(String name, HvacMode mode, SortedMap<Double, Switch> demandMap) {

        super(name, mode);

        this.demandMap = normalize(demandMap);
    }

    /**
     * Create a single switch instance.
     *
     * @param s The only switch to control the hardware. An entry of
     * {@code (theSwitch, 0)} will be put into {@link #demandMap}.
     */
    public FanPanel(String name, HvacMode mode, Switch s) {
        this(name, mode, normalize(map(s)));
    }

    public FanPanel(String name, String mode, SortedMap<Double, Switch> demandMap) {
        this(name, HvacMode.valueOf(mode), demandMap);
    }

    public FanPanel(String name, String mode, Switch s) {
        this(name, mode, normalize(map(s)));
    }

    public FanPanel(String name, String mode, SortedMap<Double, Switch> demandMap, ZoneController zc) {

        this(name, HvacMode.valueOf(mode), demandMap);

        zc.addConsumer(this);
    }

    public FanPanel(String name, String mode, Switch s, ZoneController zc) {
        this(name, mode, normalize(map(s)), zc);
    }

    @Override
    public JmxDescriptor getJmxDescriptor() {

        return new JmxDescriptor(
                "dz",
                "Fan Panel: " + getName(),
                Integer.toHexString(hashCode()),
                "Controls fan panel");
    }

    @Override
    public String toString() {

        // The only essential parts (for now) are the name and the demand map,
        // everything else is secondary

        // VT: NOTE: It might be a good idea to package this as JSON
        return getClass().getSimpleName() + ": {name:" + getName() + ", demandMap:" + demandMap + "}";
    }

    @Override
    protected Future<Status> set(HvacMode mode, double demand, double speed) {

        Callable<Status> c = new Callable<Status>() {

            @Override
            public Status call() throws Exception {
                ThreadContext.push("set");
                try {

                    double signal = map(demand, speed);

                    logger.debug("signal: ({}, {}) => {}", demand, speed, signal);

                    execute(signal, demandMap);

                    Status statusActual = new Status(mode, demand, speed);

                    setStatusActual(statusActual);

                    return statusActual;

                } finally {
                    ThreadContext.pop();
                }
            }
        };

        return executor.submit(c);
    }

    @Override
    protected void verifyModeChange(HvacMode mode) {

        if ((this.mode == HvacMode.COOLING && mode == HvacMode.HEATING)
         || (this.mode == HvacMode.HEATING && mode == HvacMode.COOLING)) {

                   throw new UnsupportedOperationException("Mode change is not supported by this device");
               }
    }
}
